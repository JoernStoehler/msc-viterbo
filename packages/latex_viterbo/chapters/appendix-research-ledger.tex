\end{itemize}

\section{Completed Research Questions}

\subsection{E0: Computational Validation of HK--O Counterexample}
\label{ledger:e0}

\textbf{Research question:} Can we computationally reproduce the Haim-Kislev--Ostrover 2024 counterexample and verify it violates Viterbo's conjecture?

\textbf{Specification:} Implement the polytope from \cite{HaimKislevOstrover2024}, compute $\cEHZ(K)$ using the tube algorithm, and verify $\cEHZ(K) < \sqrt{\vol(K)}$.

\textbf{Result:} Yes, confirmed. The polytope violates Viterbo with $\cEHZ(K) / \sqrt{\vol(K)} \approx 0.98$.

\textbf{Data:} See \texttt{packages/python\_viterbo/data/counterexamples/hk-o-2024/}

\textbf{Thesis section:} Chapter~\ref{chap:counterexample}

\textbf{Related:} GitHub Issue \#32 (CLOSED)
